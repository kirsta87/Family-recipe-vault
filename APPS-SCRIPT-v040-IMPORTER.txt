const SHEET = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
const FAMILY_KEY = "KEEP-YOUR-EXISTING-FAMILY-KEY-HERE";

const HEADERS = [
  "name","url","id","source","image","protein","type","cuisine","tags",
  "prep_time","cook_time","total_time","kirsta_rating","tj_rating",
  "torrin_rating","torrin_notes","notes","made_count","hidden","added",
  "last_made","ingredients","instructions","nutrition"
];

function doGet() {
  return jsonResponse({
    success: true,
    message: "Recipe Vault bridge is working."
  });
}

function doPost(e) {
  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(30000);

    const data = readRequest(e);

    if (data.key !== FAMILY_KEY) {
      return jsonResponse({
        success: false,
        error: "Unauthorized"
      });
    }

    if (data.action === "import") {
      return jsonResponse(importHelloFreshRecipe(data.url));
    }

    if (data.action === "add") {
      return jsonResponse(addRecipe(data));
    }

    if (data.action === "update") {
      return jsonResponse(updateRecipe(data));
    }

    return jsonResponse({
      success: false,
      error: "Unknown action"
    });

  } catch (error) {
    return jsonResponse({
      success: false,
      error: error.message
    });

  } finally {
    lock.releaseLock();
  }
}

function readRequest(e) {
  if (e && e.parameter && e.parameter.payload) {
    return JSON.parse(e.parameter.payload);
  }

  return {};
}

function importHelloFreshRecipe(url) {
  verifyHeaders();

  if (!url || !/^https:\/\/(www\.)?hellofresh\.com\/recipes\//i.test(url)) {
    return {
      success: false,
      error: "That is not a supported HelloFresh recipe URL."
    };
  }

  const idMatch = String(url).match(/-([a-f0-9]{24})(?:[/?#]|$)/i);

  if (!idMatch) {
    return {
      success: false,
      error: "The HelloFresh recipe ID could not be found."
    };
  }

  const recipeId = idMatch[1];

  if (findRecipeRow(recipeId, url)) {
    return {
      success: false,
      error: "That recipe is already in the vault."
    };
  }

  const response = UrlFetchApp.fetch(url, {
    muteHttpExceptions: true,
    followRedirects: true,
    headers: {
      "User-Agent": "Mozilla/5.0 RecipeVault/1.0"
    }
  });

  const status = response.getResponseCode();

  if (status !== 200) {
    return {
      success: false,
      error: "HelloFresh returned HTTP " + status + "."
    };
  }

  const html = response.getContentText();
  const recipe = parseHelloFreshPage(html, url, recipeId);

  if (!recipe.name || !recipe.ingredients.length || !recipe.instructions.length) {
    return {
      success: false,
      error: "The recipe page loaded, but its structured recipe details could not be extracted."
    };
  }

  appendRecipe(recipe);

  return {
    success: true,
    action: "imported",
    recipe: recipe,
    row: SHEET.getLastRow()
  };
}

function parseHelloFreshPage(html, url, recipeId) {
  const jsonLdRecipe = findRecipeJsonLd(html);

  if (jsonLdRecipe) {
    return recipeFromJsonLd(jsonLdRecipe, url, recipeId);
  }

  return recipeFromPageFallback(html, url, recipeId);
}

function findRecipeJsonLd(html) {
  const scripts = html.match(
    /<script[^>]+type=["']application\/ld\+json["'][^>]*>[\s\S]*?<\/script>/gi
  ) || [];

  for (let index = 0; index < scripts.length; index++) {
    const content = scripts[index]
      .replace(/^<script[^>]*>/i, "")
      .replace(/<\/script>$/i, "")
      .trim();

    try {
      const parsed = JSON.parse(decodeHtmlEntities(content));
      const candidates = Array.isArray(parsed)
        ? parsed
        : parsed["@graph"]
          ? parsed["@graph"]
          : [parsed];

      for (let candidateIndex = 0; candidateIndex < candidates.length; candidateIndex++) {
        const candidate = candidates[candidateIndex];
        const types = Array.isArray(candidate["@type"])
          ? candidate["@type"]
          : [candidate["@type"]];

        if (types.indexOf("Recipe") !== -1) {
          return candidate;
        }
      }
    } catch (error) {
      // Continue to the next JSON-LD script.
    }
  }

  return null;
}

function recipeFromJsonLd(data, url, recipeId) {
  const ingredients = normalizeStringArray(data.recipeIngredient);
  const instructions = normalizeInstructions(data.recipeInstructions);
  const nutrition = formatNutrition(data.nutrition);

  const totalMinutes = durationToMinutes(data.totalTime);
  const prepMinutes = durationToMinutes(data.prepTime);
  const cookMinutes = durationToMinutes(data.cookTime);

  return {
    name: cleanText(data.name) || titleFromUrl(url),
    url: url,
    id: recipeId,
    source: "HelloFresh",
    image: firstImage(data.image),
    protein: detectProtein(
      [
        data.name,
        data.description,
        ingredients.join(" ")
      ].join(" ")
    ),
    type: detectMealType(data.name || ""),
    cuisine: cleanText(
      Array.isArray(data.recipeCuisine)
        ? data.recipeCuisine.join("|")
        : data.recipeCuisine
    ),
    tags: normalizeKeywords(data.keywords),
    prep_time: prepMinutes || "",
    cook_time: cookMinutes || "",
    total_time: totalMinutes || "",
    kirsta_rating: "",
    tj_rating: "",
    torrin_rating: "",
    torrin_notes: "",
    notes: "",
    made_count: 0,
    hidden: false,
    added: todayString(),
    last_made: "",
    ingredients: ingredients,
    instructions: instructions,
    nutrition: nutrition
  };
}

function recipeFromPageFallback(html, url, recipeId) {
  const text = htmlToText(html);

  const name =
    firstMatch(html, /<h1[^>]*>([\s\S]*?)<\/h1>/i) ||
    titleFromUrl(url);

  const totalTime =
    firstMatch(text, /Total\s*:?\s*(\d+)\s*minutes?/i) ||
    firstMatch(text, /(\d+)\s*minutes?\s*Difficulty/i);

  const ingredientBlock = between(
    text,
    "Ingredients",
    "Nutrition Values"
  );

  const instructionBlock = between(
    text,
    "Cooking Steps",
    "Customer reviews"
  );

  return {
    name: cleanText(stripTags(name)),
    url: url,
    id: recipeId,
    source: "HelloFresh",
    image: firstMatch(
      html,
      /<meta[^>]+property=["']og:image["'][^>]+content=["']([^"']+)["']/i
    ),
    protein: detectProtein(name + " " + ingredientBlock),
    type: detectMealType(name),
    cuisine: "",
    tags: "",
    prep_time: "",
    cook_time: "",
    total_time: totalTime ? Number(totalTime) : "",
    kirsta_rating: "",
    tj_rating: "",
    torrin_rating: "",
    torrin_notes: "",
    notes: "",
    made_count: 0,
    hidden: false,
    added: todayString(),
    last_made: "",
    ingredients: fallbackIngredients(ingredientBlock),
    instructions: fallbackInstructions(instructionBlock),
    nutrition: fallbackNutrition(text)
  };
}

function normalizeStringArray(value) {
  if (!value) return [];

  const array = Array.isArray(value) ? value : [value];

  return array
    .map(item => cleanText(item))
    .filter(Boolean);
}

function normalizeInstructions(value) {
  if (!value) return [];

  const output = [];

  function visit(item) {
    if (!item) return;

    if (typeof item === "string") {
      const text = cleanText(item);
      if (text) output.push(text);
      return;
    }

    if (Array.isArray(item)) {
      item.forEach(visit);
      return;
    }

    if (item.itemListElement) {
      visit(item.itemListElement);
      return;
    }

    const text = cleanText(item.text || item.name);
    if (text) output.push(text);
  }

  visit(value);
  return output;
}

function formatNutrition(nutrition) {
  if (!nutrition) return "";

  const fields = [
    ["Calories", nutrition.calories],
    ["Protein", nutrition.proteinContent],
    ["Carbohydrates", nutrition.carbohydrateContent],
    ["Fat", nutrition.fatContent],
    ["Saturated fat", nutrition.saturatedFatContent],
    ["Fiber", nutrition.fiberContent],
    ["Sugar", nutrition.sugarContent],
    ["Sodium", nutrition.sodiumContent],
    ["Cholesterol", nutrition.cholesterolContent]
  ];

  return fields
    .filter(item => item[1])
    .map(item => item[0] + ": " + cleanText(item[1]))
    .join(" | ");
}

function firstImage(image) {
  if (!image) return "";

  if (typeof image === "string") return image;
  if (Array.isArray(image)) return firstImage(image[0]);
  if (image.url) return image.url;

  return "";
}

function normalizeKeywords(value) {
  if (!value) return "";

  if (Array.isArray(value)) {
    return value.map(cleanText).filter(Boolean).join("|");
  }

  return cleanText(value)
    .split(",")
    .map(item => item.trim())
    .filter(Boolean)
    .join("|");
}

function durationToMinutes(value) {
  if (!value) return "";

  const match = String(value).match(
    /^P(?:(\d+)D)?(?:T(?:(\d+)H)?(?:(\d+)M)?)?$/i
  );

  if (!match) {
    const number = String(value).match(/\d+/);
    return number ? Number(number[0]) : "";
  }

  return (
    (Number(match[1]) || 0) * 1440 +
    (Number(match[2]) || 0) * 60 +
    (Number(match[3]) || 0)
  );
}

function detectProtein(value) {
  const text = String(value || "").toLowerCase();

  if (text.indexOf("chicken") !== -1) return "Chicken";
  if (
    text.indexOf("beef") !== -1 ||
    text.indexOf("steak") !== -1
  ) return "Beef";
  if (
    text.indexOf("pork") !== -1 ||
    text.indexOf("sausage") !== -1 ||
    text.indexOf("bacon") !== -1
  ) return "Pork";
  if (text.indexOf("turkey") !== -1) return "Turkey";
  if (
    text.indexOf("shrimp") !== -1 ||
    text.indexOf("salmon") !== -1 ||
    text.indexOf("fish") !== -1
  ) return "Seafood";
  if (
    text.indexOf("tofu") !== -1 ||
    text.indexOf("vegetarian") !== -1 ||
    text.indexOf("chickpea") !== -1
  ) return "Vegetarian";

  return "Other";
}

function detectMealType(value) {
  const text = String(value || "").toLowerCase();

  if (text.indexOf("taco") !== -1) return "Tacos";
  if (
    text.indexOf("pasta") !== -1 ||
    text.indexOf("spaghetti") !== -1 ||
    text.indexOf("rigatoni") !== -1 ||
    text.indexOf("ravioli") !== -1
  ) return "Pasta";
  if (text.indexOf("burger") !== -1) return "Burgers";
  if (text.indexOf("bowl") !== -1) return "Bowls";
  if (
    text.indexOf("flatbread") !== -1 ||
    text.indexOf("pizza") !== -1
  ) return "Flatbread";
  if (
    text.indexOf("soup") !== -1 ||
    text.indexOf("stew") !== -1 ||
    text.indexOf("chowder") !== -1
  ) return "Soup";
  if (
    text.indexOf("sandwich") !== -1 ||
    text.indexOf("sando") !== -1 ||
    text.indexOf("melt") !== -1
  ) return "Sandwiches";

  return "Other";
}

function fallbackIngredients(block) {
  if (!block) return [];

  const cleaned = block
    .replace(/Serving amount[\s\S]*?(?=\d|Salt|Pepper)/i, "")
    .replace(/Not included in your delivery/gi, "\n")
    .split(/\n+/)
    .map(cleanText)
    .filter(Boolean);

  const output = [];

  for (let index = 0; index < cleaned.length - 1; index++) {
    const quantity = cleaned[index];
    const ingredient = cleaned[index + 1];

    if (
      /^(\d|¼|½|¾)/.test(quantity) &&
      !/^(\d|¼|½|¾)/.test(ingredient)
    ) {
      output.push(quantity + " " + ingredient);
      index++;
    }
  }

  ["Salt", "Pepper", "Cooking Oil"].forEach(item => {
    if (block.indexOf(item) !== -1) output.push(item);
  });

  return output;
}

function fallbackInstructions(block) {
  if (!block) return [];

  const matches = block.match(
    /(?:^|\n)\s*\d+\s*\n([\s\S]*?)(?=(?:\n\s*\d+\s*\n)|$)/g
  ) || [];

  return matches
    .map(item => cleanText(item.replace(/^\s*\d+\s*/,"")))
    .filter(Boolean);
}

function fallbackNutrition(text) {
  const block = between(
    text,
    "Nutrition Values",
    "Utensils"
  );

  return cleanText(block)
    .replace(/\s*\/ per serving\s*/i, "")
    .replace(/\s+/g, " | ");
}

function appendRecipe(recipe) {
  const row = HEADERS.map(header => {
    if (header === "ingredients") {
      return JSON.stringify(recipe.ingredients || []);
    }

    if (header === "instructions") {
      return JSON.stringify(recipe.instructions || []);
    }

    return recipe[header] ?? "";
  });

  SHEET.appendRow(row);
}

function addRecipe(data) {
  verifyHeaders();

  const recipe = data.updates || {};

  if (findRecipeRow(recipe.id, recipe.url)) {
    return {
      success: false,
      error: "That recipe is already in the vault."
    };
  }

  appendRecipe(recipe);

  return {
    success: true,
    action: "added",
    row: SHEET.getLastRow()
  };
}

function updateRecipe(data) {
  verifyHeaders();

  const row = findRecipeRow(data.id, data.url);

  if (!row) {
    return {
      success: false,
      error: "Recipe not found."
    };
  }

  const headerMap = getHeaderMap();
  const updates = data.updates || {};

  Object.keys(updates).forEach(key => {
    if (headerMap[key]) {
      let value = updates[key];

      if (
        (key === "ingredients" || key === "instructions") &&
        Array.isArray(value)
      ) {
        value = JSON.stringify(value);
      }

      SHEET
        .getRange(row, headerMap[key])
        .setValue(value);
    }
  });

  return {
    success: true,
    action: "updated",
    row: row
  };
}

function findRecipeRow(id, url) {
  const lastRow = SHEET.getLastRow();

  if (lastRow < 2) return null;

  const map = getHeaderMap();

  const values = SHEET
    .getRange(
      2,
      1,
      lastRow - 1,
      SHEET.getLastColumn()
    )
    .getValues();

  for (let index = 0; index < values.length; index++) {
    const rowId = map.id
      ? String(values[index][map.id - 1])
      : "";

    const rowUrl = map.url
      ? String(values[index][map.url - 1])
      : "";

    if (
      (id && rowId === String(id)) ||
      (url && rowUrl === String(url))
    ) {
      return index + 2;
    }
  }

  return null;
}

function getHeaderMap() {
  const headers = SHEET
    .getRange(
      1,
      1,
      1,
      SHEET.getLastColumn()
    )
    .getValues()[0];

  const map = {};

  headers.forEach((header, index) => {
    map[String(header).trim()] = index + 1;
  });

  return map;
}

function verifyHeaders() {
  const actual = SHEET
    .getRange(
      1,
      1,
      1,
      HEADERS.length
    )
    .getValues()[0]
    .map(value => String(value).trim());

  const missing = HEADERS.filter(
    header => actual.indexOf(header) === -1
  );

  if (missing.length) {
    throw new Error(
      "Missing columns: " + missing.join(", ")
    );
  }
}

function titleFromUrl(url) {
  const slug = String(url)
    .split("/")
    .filter(Boolean)
    .pop()
    .replace(/-[a-f0-9]{24}$/i, "");

  return slug
    .split("-")
    .map(word => word
      ? word.charAt(0).toUpperCase() + word.slice(1)
      : ""
    )
    .join(" ");
}

function htmlToText(html) {
  return decodeHtmlEntities(
    String(html)
      .replace(/<script[\s\S]*?<\/script>/gi, " ")
      .replace(/<style[\s\S]*?<\/style>/gi, " ")
      .replace(/<br\s*\/?>/gi, "\n")
      .replace(/<\/(p|div|li|h1|h2|h3|section)>/gi, "\n")
      .replace(/<[^>]+>/g, " ")
      .replace(/[ \t]+/g, " ")
      .replace(/\n\s+/g, "\n")
  );
}

function stripTags(value) {
  return String(value || "").replace(/<[^>]+>/g, " ");
}

function cleanText(value) {
  return decodeHtmlEntities(
    String(value || "")
      .replace(/<[^>]+>/g, " ")
      .replace(/\s+/g, " ")
      .trim()
  );
}

function decodeHtmlEntities(value) {
  return String(value || "")
    .replace(/&amp;/g, "&")
    .replace(/&quot;/g, '"')
    .replace(/&#39;/g, "'")
    .replace(/&apos;/g, "'")
    .replace(/&lt;/g, "<")
    .replace(/&gt;/g, ">")
    .replace(/&nbsp;/g, " ");
}

function firstMatch(value, regex) {
  const match = String(value || "").match(regex);
  return match ? match[1] : "";
}

function between(value, start, end) {
  const text = String(value || "");
  const startIndex = text.indexOf(start);

  if (startIndex === -1) return "";

  const contentStart = startIndex + start.length;
  const endIndex = text.indexOf(end, contentStart);

  return endIndex === -1
    ? text.slice(contentStart)
    : text.slice(contentStart, endIndex);
}

function todayString() {
  return Utilities.formatDate(
    new Date(),
    Session.getScriptTimeZone(),
    "yyyy-MM-dd"
  );
}

function jsonResponse(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
